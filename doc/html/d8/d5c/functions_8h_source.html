40 /**
41  * @brief Represents a mathematical function in a syntax tree.
42  *
43  * Function nodes in syntax trees can both be well known functions like
44  * cosine (cos) and also user defined functions. The function node represents
45  * a function call and not a function definition.
46  *
47  * A formal description of mathematical functions can be found at Wikipedia:
48  * http://en.wikipedia.org/wiki/Function_(mathematics)
49  *
50  * When instantiating a function node an argument must be supplied. The argument
51  * is a pointer to an expression node representing the value which should be used
52  * when computing the function value.
53  *
54  */
55 class FunctionNode : public ExpressionNode {
56 public:
57  FunctionNode(ExpressionNode* expression);
58  ~FunctionNode();
59 
60  int GetPrecedence();
61  char* GetText();
62  virtual SyntaxNode* GetNext();
63  virtual void Attach(SyntaxNode *node);
64  virtual void Detach(SyntaxNode *node);
65  virtual void Replace(SyntaxNode *n, SyntaxNode *x);
66 
67 protected:
68  ExpressionNode* expression;
69 };
