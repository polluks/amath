92 void TestProgram::PerformTest(const char* input, const char* result, bool show)
93 {
94  Evaluator *evaluator = new Evaluator(input);
95  evaluator->Evaluate();
96  char *res = evaluator->GetResult();
97 
98  CharBuffer *buf = new CharBuffer();
99  buf->ClearAndCopy(res);
100  buf->RemoveTrailing(NEWLINE);
101  delete evaluator;
102 
103  if (buf->Is(result)) {
104  pass++;
105  printf("PASS: [%s]" NEWLINE, show ? result : input);
106  } else {
107  fail++;
108  printf("FAIL: [%s] expected [%s] but got [%s]" NEWLINE, input, result, buf->GetString());
109  }
110 
111  delete buf;
112 }
