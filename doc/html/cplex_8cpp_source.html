81 int ComplexNumber::GetPrecedence()
82 {
83  if ((creal(z) < 0.0) || (creal(z) == 0.0 && cimag(z) < 0.0)) {
84  return -1;
85  } else if (creal(z) != 0.0 && cimag(z) != 0.0) {
86  return 2;
87  } else {
88  return 0;
89  }
90 }
