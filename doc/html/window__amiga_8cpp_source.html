61 void AmigaWindow::Create()
62 {
63  NewWindow nw = {
64  10, 20, // left, top
65  WWIDTH, WHEIGHT, // width, height
66  (UBYTE)-1, (UBYTE)-1, // detailpen, blockpen
67  IDCMP_CLOSEWINDOW |
68  IDCMP_MENUPICK,
69  WFLG_SIZEGADGET |
70  WFLG_DRAGBAR |
71  WFLG_DEPTHGADGET |
72  WFLG_CLOSEGADGET |
73  WFLG_ACTIVATE |
74  WFLG_SIMPLE_REFRESH,
75  NULL, NULL, // user gadgets, user checkmark
76  (UBYTE*) TXTTITLE, // title
77  NULL, NULL, // window screen, super bitmap
78  400, 75, // min width, height
79  1600, 1050, // max width, height
80  WBENCHSCREEN // open on workbench screen
81  };
82 
83  succeed = true;
84  succeed = succeed && (writeport = CreatePort(PORTCR, 0)) != 0;
85  succeed = succeed && (readport = CreatePort(PORTCW, 0)) != 0;
86  succeed = succeed && (writereq.st = CreateExtIO(writeport, sizeof(IOStdReq))) != 0;
87  succeed = succeed && (readreq.st = CreateExtIO(readport, sizeof(IOStdReq))) != 0;
88  succeed = succeed && (window = OpenWindow(&nw)) != 0;
89  if (!succeed) return;
90 
91  windowsig = 1 << window->UserPort->mp_SigBit;
92  writereq.io->io_Data = (APTR) window;
93  writereq.io->io_Length = sizeof(Window);
94 
95  openconsole = OpenDevice(DEVCONSOLE, CONU_SNIPMAP, writereq.st, CONFLAG_DEFAULT) == 0;
96  succeed = openconsole;
97  if (!openconsole) return;
98 
99  readreq.st->io_Device = writereq.st->io_Device;
100  readreq.st->io_Unit = writereq.st->io_Unit;
101 
102  // Queue input
103  ReadChar(readreq);
104 
105  // Do GUI stuff
106  //menu = new AmigaMenu(window);
107  //menu->Attach();
108 }

153 void AmigaWindow::Run()
154 {
155  Create();
156  StartMessage();
157  exit = false;
158 
159  while(!exit) {
160  Prompt();
161  ReadLine();
162  Evaluator *evaluator = new Evaluator(line);
163  evaluator->Evaluate();
164  const char *out = evaluator->GetResult();
165  WriteString(out, StrLen(out));
166  delete evaluator;
167  }
168 
169  Cleanup();
170 }

177 void AmigaWindow::ReadLine()
178 {
179  unsigned long conreadsig = 1L << readport->mp_SigBit;
180  proc->StartInput();
181 
182  while(!proc->InputDone() && !exit)
183  {
184  unsigned long signals = Wait(conreadsig | windowsig);
185 
186  if (signals & conreadsig)
187  {
188  unsigned char ch = TryGetChar(readport);
189  if (ch != 0)
190  {
191  const char *out = proc->ProcessChar(ch);
192  WriteString(out);
193  }
194  }
195 
196  if (signals & windowsig)
197  {
198  STMessage w;
199  while ((w.msg = GetMsg(window->UserPort)))
200  {
201  switch(w.imsg->Class)
202  {
203  case IDCMP_CLOSEWINDOW:
204  exit = true;
205  break;
206  //case IDCMP_MENUPICK:
207  // menu->Process();
208  // break;
209  default:
210  break;
211  }
212 
213  ReplyMsg(w.msg);
214  }
215  }
216  }
217 
218  line = proc->GetLine();
219 }
