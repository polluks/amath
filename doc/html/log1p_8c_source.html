46 /* double log1p(double x)
47  *
48  * Method :
49  * 1. Argument Reduction: find k and f such that
50  * 1+x = 2^k * (1+f),
51  * where sqrt(2)/2 < 1+f < sqrt(2) .
52  *
53  * Note. If k=0, then f=x is exact. However, if k!=0, then f
54  * may not be representable exactly. In that case, a correction
55  * term is need. Let u=1+x rounded. Let c = (1+x)-u, then
56  * log(1+x) - log(u) ~ c/u. Thus, we proceed to compute log(u),
57  * and add back the correction term c/u.
58  * (Note: when x > 2**53, one can simply return log(x))
59  *
60  * 2. Approximation of log1p(f).
61  * Let s = f/(2+f) ; based on log(1+f) = log(1+s) - log(1-s)
62  * = 2s + 2/3 s**3 + 2/5 s**5 + .....,
63  * = 2s + s*R
64  * We use a special Remes algorithm on [0,0.1716] to generate
65  * a polynomial of degree 14 to approximate R The maximum error
66  * of this polynomial approximation is bounded by 2**-58.45. In
67  * other words,
68  * 2 4 6 8 10 12 14
69  * R(z) ~ Lp1*s +Lp2*s +Lp3*s +Lp4*s +Lp5*s +Lp6*s +Lp7*s
70  * (the values of Lp1 to Lp7 are listed in the program)
71  * and
72  * | 2 14 | -58.45
73  * | Lp1*s +...+Lp7*s - R(z) | <= 2
74  * | |
75  * Note that 2s = f - s*f = f - hfsq + s*hfsq, where hfsq = f*f/2.
76  * In order to guarantee error in log below 1ulp, we compute log
77  * by
78  * log1p(f) = f - (hfsq - s*(hfsq+R)).
79  *
80  * 3. Finally, log1p(x) = k*ln2 + log1p(f).
81  * = k*ln2_hi+(f-(hfsq-(s*(hfsq+R)+k*ln2_lo)))
82  * Here ln2 is split into two floating point number:
83  * ln2_hi + ln2_lo,
84  * where n*ln2_hi is always exact for |n| < 2000.
85  *
86  * Special cases:
87  * log1p(x) is NaN with signal if x < -1 (including -INF) ;
88  * log1p(+INF) is +INF; log1p(-1) is -INF with signal;
89  * log1p(NaN) is that NaN with no signal.
90  *
91  * Accuracy:
92  * according to an error analysis, the error is always less than
93  * 1 ulp (unit in the last place).
94  *
95  * Constants:
96  * The hexadecimal values are the intended ones for the following
97  * constants. The decimal values may be used, provided that the
98  * compiler will convert from decimal to binary accurately enough
99  * to produce the hexadecimal values shown.
100  *
101  * Note: Assuming log() return accurate answer, the following
102  * algorithm can be used to compute log1p(x) to within a few ULP:
103  *
104  * u = 1+x;
105  * if(u==1.0) return x ; else
106  * return log(u)*(x/(u-1.0));
107  *
108  * See HP-15C Advanced Functions Handbook, p.193.
109  */

125 double log1p(double x)
126 {
127  double hfsq,f,c,s,z,R,u;
128  sword k,hx,hu,ax;
129 
130  f = 0.0;
131  c = 0.0;
132  hu = 0;
133 
134  GET_HIGH_WORD(hx,x); /* high word of x */
135  ax = hx&0x7fffffff;
136 
137  k = 1;
138  if (hx < 0x3FDA827A) { /* x < 0.41422 */
139  if(ax>=0x3ff00000) { /* x <= -1.0 */
140  if(x==-1.0) return -two54/zero; /* log1p(-1)=+inf */
141  else return (x-x)/(x-x); /* log1p(x<-1)=NaN */
142  }
143  if(ax<0x3e200000) { /* |x| < 2**-29 */
144  if(two54+x>zero /* raise inexact */
145  &&ax<0x3c900000) /* |x| < 2**-54 */
146  return x;
147  else
148  return x - x*x*0.5;
149  }
150  if(hx>0||hx<=((int)0xbfd2bec3)) {
151  k=0;
152  f=x;
153  hu=1;
154  } /* -0.2929<x<0.41422 */
155  }
156  if (hx >= 0x7ff00000) return x+x;
157  if(k!=0) {
158  if(hx<0x43400000) {
159  u = 1.0+x;
160  GET_HIGH_WORD(hu,u); /* high word of u */
161  k = (hu>>20)-1023;
162  c = (k>0)? 1.0-(u-x):x-(u-1.0);/* correction term */
163  c /= u;
164  } else {
165  u = x;
166  GET_HIGH_WORD(hu,u); /* high word of u */
167  k = (hu>>20)-1023;
168  c = 0;
169  }
170  hu &= 0x000fffff;
171  if(hu<0x6a09e) {
172  SET_HIGH_WORD(u, hu|0x3ff00000); /* normalize u */
173  } else {
174  k += 1;
175  SET_HIGH_WORD(u, hu|0x3fe00000); /* normalize u/2 */
176  hu = (0x00100000-hu)>>2;
177  }
178  f = u-1.0;
179  }
180  hfsq=0.5*f*f;
181  if(hu==0) { /* |f| < 2**-20 */
182  if(f==zero) {
183  if(k==0) return zero;
184  else {
185  c += k*ln2_lo;
186  return k*ln2_hi+c;
187  }
188  }
189  R = hfsq*(1.0-0.66666666666666666*f);
190  if(k==0) return f-R;
191  else
192  return k*ln2_hi-((R-(k*ln2_lo+c))-f);
193  }
194  s = f/(2.0+f);
195  z = s*s;
196  R = z*(Lp1+z*(Lp2+z*(Lp3+z*(Lp4+z*(Lp5+z*(Lp6+z*Lp7))))));
197  if(k==0) return f-(hfsq-s*(hfsq+R));
198  else
199  return k*ln2_hi-((hfsq-(s*(hfsq+R)+(k*ln2_lo+c)))-f);
200 }
