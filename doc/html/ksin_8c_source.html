54 /**
55  * @brief Kernel sin function.
56  * @version 1.3
57  * @date 95/01/18
58  * @details
59  *
60  * Kernel sin function on [-pi/4, pi/4], pi/4 ~ 0.7854
61  * Input x is assumed to be bounded by ~pi/4 in magnitude.
62  * Input y is the tail of x.
63  * Input iy indicates whether y is 0. (if iy=0, y assume to be 0).
64  *
65  * Algorithm
66  * 1. Since sin(-x) = -sin(x), we need only to consider positive x.
67  * 2. if x < 2^-27 (hx<0x3e400000 0), return x with inexact if x!=0.
68  * 3. sin(x) is approximated by a polynomial of degree 13 on
69  * [0,pi/4]
70  * 3 13
71  * sin(x) ~ x + S1*x + ... + S6*x
72  * where
73  *
74  * |sin(x) 2 4 6 8 10 12 | -58
75  * |----- - (1+S1*x +S2*x +S3*x +S4*x +S5*x +S6*x )| <= 2
76  * | x |
77  *
78  * 4. sin(x+y) = sin(x) + sin'(x')*y
79  * ~ sin(x) + (1-x*x/2)*y
80  * For better accuracy, let
81  * 3 2 2 2 2
82  * r = x *(S2+x *(S3+x *(S4+x *(S5+x *S6))))
83  * then 3 2
84  * sin(x) = x + (S1*x + (x *(r-y/2)+y))
85  * </pre>
86  * @copyright Copyright (C) 1993 by Sun Microsystems, Inc. All rights reserved.
87  * @license Developed at SunSoft, a Sun Microsystems, Inc. business. Permission
88  * to use, copy, modify, and distribute this software is freely granted,
89  * provided that this notice is preserved.
90  */

92 double __kernel_sin(double x, double y, int iy)
93 {
94  double z,r,v;
95  sword ix;
96  GET_HIGH_WORD(ix,x);
97  ix &=0x7fffffff; /* high word of x */
98  if(ix<0x3e400000) /* |x| < 2**-27 */
99  {
100  if((int)x==0) return x; /* generate inexact */
101  }
102  z = x*x;
103  v = z*x;
104  r = S2+z*(S3+z*(S4+z*(S5+z*S6)));
105  if(iy==0) return x+v*(S1+z*r);
106  else return x-((z*(half*y-v*r)-y)-v*S1);
107 }
