53 /**
54  * @brief Hyperbolic cosine function.
55  * @version 1.3
56  * @date 95/01/18
57  * @details
58  * <pre>
59  * Method :
60  * mathematically cosh(x) if defined to be (exp(x)+exp(-x))/2
61  * 1. Replace x by |x| (cosh(x) = cosh(-x)).
62  * 2.
63  * [ exp(x) - 1 ]^2
64  * 0 <= x <= ln2/2 : cosh(x) := 1 + -------------------
65  * 2*exp(x)
66  *
67  * exp(x) + 1/exp(x)
68  * ln2/2 <= x <= 22 : cosh(x) := -------------------
69  * 2
70  * 22 <= x <= lnovft : cosh(x) := exp(x)/2
71  * lnovft <= x <= ln2ovft: cosh(x) := exp(x/2)/2 * exp(x/2)
72  * ln2ovft < x : cosh(x) := huge*huge (overflow)
73  *
74  * Special cases:
75  * cosh(x) is |x| if x is +INF, -INF, or NaN.
76  * only cosh(0)=1 is exact for finite x.
77  * </pre>
78  * @copyright Copyright (C) 1993 by Sun Microsystems, Inc. All rights reserved.
79  * @license Developed at SunSoft, a Sun Microsystems, Inc. business. Permission
80  * to use, copy, modify, and distribute this software is freely granted,
81  * provided that this notice is preserved.
82  */

84 double cosh(double x)
85 {
86  double t,w;
87  sword ix;
88  uword lx;
89 
90  /* High word of |x|. */
91  GET_HIGH_WORD(ix,x);
92  ix &= 0x7fffffff;
93 
94  /* x is INF or NaN */
95  if(ix>=0x7ff00000) return x*x;
96 
97  /* |x| in [0,0.5*ln2], return 1+expm1(|x|)^2/(2*exp(|x|)) */
98  if(ix<0x3fd62e43) {
99  t = expm1(fabs(x));
100  w = one+t;
101  if (ix<0x3c800000) return w; /* cosh(tiny) = 1 */
102  return one+(t*t)/(w+w);
103  }
104 
105  /* |x| in [0.5*ln2,22], return (exp(|x|)+1/exp(|x|)/2; */
106  if (ix < 0x40360000) {
107  t = exp(fabs(x));
108  return half*t+half/t;
109  }
110 
111  /* |x| in [22, log(maxdouble)] return half*exp(|x|) */
112  if (ix < 0x40862E42) return half*exp(fabs(x));
113 
114  /* |x| in [log(maxdouble), overflowthresold] */
115  lx = *( (((*(unsigned*)&one)>>29)) + (unsigned*)&x);
116  if (ix<0x408633CE ||
117  ((ix==0x408633ce)&&(lx<=(unsigned)0x8fb9f87d))) {
118  w = exp(half*fabs(x));
119  t = half*w;
120  return t*w;
121  }
122 
123  /* |x| > overflowthresold, cosh(x) overflow */
124  return huge*huge;
125 }
