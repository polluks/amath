146 /**
147  * @brief Multiplication of two complex numbers.
148  *
149  */
150 complex cmul(complex y, complex z)
151 {
152  complex w;
153  double a, b, c, d;
154 
155  // (a+bi)(c+di)
156  a = creal(y);
157  b = cimag(y);
158  c = creal(z);
159  d = cimag(z);
160 
161  // (ac -bd) + (ad + bc)i
162  w = cpack(a * c - b * d, a * d + b * c);
163  return w;
164 }

166 /**
167  * @brief Division of two complex numbers.
168  *
169  */
170 complex cdiv(complex y, complex z)
171 {
172  complex w;
173  double a, b, c, d;
174  double q, v, x;
175 
176  a = creal(y);
177  b = cimag(y);
178  c = creal(z);
179  d = cimag(z);
180 
181  q = c * c + d * d;
182  v = a * c + b * d;
183  x = b * c - a * d;
184 
185  w = cpack(v / q, x / q);
186  return w;
187 }

211 {

213 
214  if (fabs(x) <= 0.5) {
215  *c = cosh(x);
