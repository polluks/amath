55 /**
56  * @brief Kernel cos function.
57  * @version 1.4
58  * @date 96/03/07
59  * @details
60  *
61  * <pre>
62  * Kernel cos function on [-pi/4, pi/4], pi/4 ~ 0.785398164
63  * Input x is assumed to be bounded by ~pi/4 in magnitude.
64  * Input y is the tail of x.
65  *
66  * Algorithm
67  * 1. Since cos(-x) = cos(x), we need only to consider positive x.
68  * 2. if x < 2^-27 (hx<0x3e400000 0), return 1 with inexact if x!=0.
69  * 3. cos(x) is approximated by a polynomial of degree 14 on
70  * [0,pi/4]
71  * 4 14
72  * cos(x) ~ 1 - x*x/2 + C1*x + ... + C6*x
73  * where the Remes error is
74  *
75  * | 2 4 6 8 10 12 14 | -58
76  * |cos(x)-(1-.5*x +C1*x +C2*x +C3*x +C4*x +C5*x +C6*x )| <= 2
77  * | |
78  *
79  * 4 6 8 10 12 14
80  * 4. let r = C1*x +C2*x +C3*x +C4*x +C5*x +C6*x , then
81  * cos(x) = 1 - x*x/2 + r
82  * since cos(x+y) ~ cos(x) - sin(x)*y
83  * ~ cos(x) - x*y,
84  * a correction term is necessary in cos(x) and hence
85  * cos(x+y) = 1 - (x*x/2 - (r - x*y))
86  * For better accuracy when x > 0.3, let qx = |x|/4 with
87  * the last 32 bits mask off, and if x > 0.78125, let qx = 0.28125.
88  * Then
89  * cos(x+y) = (1-qx) - ((x*x/2-qx) - (r-x*y)).
90  * Note that 1-qx and (x*x/2-qx) is EXACT here, and the
91  * magnitude of the latter is at least a quarter of x*x/2,
92  * thus, reducing the rounding error in the subtraction.
93  * </pre>
94  * @copyright Copyright (C) 1993 by Sun Microsystems, Inc. All rights reserved.
95  * @license Developed at SunSoft, a Sun Microsystems, Inc. business. Permission
96  * to use, copy, modify, and distribute this software is freely granted,
97  * provided that this notice is preserved.
98  */

100 double __kernel_cos(double x, double y)
101 {
102  double a,hz,z,r,qx;
103  sword ix;
104  GET_HIGH_WORD(ix, x);
105  ix &= 0x7fffffff; /* ix = |x|'s high word*/
106  if(ix<0x3e400000) { /* if x < 2**27 */
107  if(((int)x)==0) return one; /* generate inexact */
108  }
109  z = x*x;
110  r = z*(C1+z*(C2+z*(C3+z*(C4+z*(C5+z*C6)))));
111  if(ix < 0x3FD33333) /* if |x| < 0.3 */
112  return one - (0.5*z - (z*r - x*y));
113  else {
114  if(ix > 0x3fe90000) { /* x > 0.78125 */
115  qx = 0.28125;
116  } else {
117  INSERT_WORDS(qx,ix-0x00200000,0);
118  }
119  hz = 0.5*z-qx;
120  a = one-qx;
121  return a - (hz - (z*r-x*y));
122  }
123 }
