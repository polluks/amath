69 /**
70  * @brief Kernel tan function.
71  * @version 1.5
72  * @date 04/04/22
73  * @details
74  *
75  * Kernel tan function on [-pi/4, pi/4], pi/4 ~ 0.7854
76  * Input x is assumed to be bounded by ~pi/4 in magnitude.
77  * Input y is the tail of x.
78  * Input k indicates whether tan (if k = 1) or -1/tan (if k = -1) is returned.
79  *
80  * Algorithm
81  * 1. Since tan(-x) = -tan(x), we need only to consider positive x.
82  * 2. if x < 2^-28 (hx<0x3e300000 0), return x with inexact if x!=0.
83  * 3. tan(x) is approximated by a odd polynomial of degree 27 on
84  * [0,0.67434]
85  * 3 27
86  * tan(x) ~ x + T1*x + ... + T13*x
87  * where
88  *
89  * |tan(x) 2 4 26 | -59.2
90  * |----- - (1+T1*x +T2*x +.... +T13*x )| <= 2
91  * | x |
92  *
93  * Note: tan(x+y) = tan(x) + tan'(x)*y
94  * ~ tan(x) + (1+x*x)*y
95  * Therefore, for better accuracy in computing tan(x+y), let
96  * 3 2 2 2 2
97  * r = x *(T2+x *(T3+x *(...+x *(T12+x *T13))))
98  * then
99  * 3 2
100  * tan(x+y) = x + (T1*x + (x *(r+y)+y))
101  *
102  * 4. For x in [0.67434,pi/4], let y = pi/4 - x, then
103  * tan(x) = tan(pi/4-y) = (1-tan(y))/(1+tan(y))
104  * = 1 - 2*(tan(y) - (tan(y)^2)/(1+tan(y)))
105  * </pre>
106  * @copyright Copyright (C) 1993 by Sun Microsystems, Inc. All rights reserved.
107  * @license Developed at SunSoft, a Sun Microsystems, Inc. business. Permission
108  * to use, copy, modify, and distribute this software is freely granted,
109  * provided that this notice is preserved.
110  */

112 double __kernel_tan(double x, double y, int iy) {
113  double z, r, v, w, s;
114  sword ix, hx;
115 
116  GET_HIGH_WORD(hx,x); /* high word of x */
117  ix = hx & 0x7fffffff; /* high word of |x| */
118  if (ix < 0x3e300000) { /* x < 2**-28 */
119  if ((int) x == 0) { /* generate inexact */
120  uword low;
121  GET_LOW_WORD(low,x);
122  if (((ix | low) | (iy + 1)) == 0)
123  return one / fabs(x);
124  else {
125  if (iy == 1)
126  return x;
127  else { /* compute -1 / (x+y) carefully */
128  double a, t;
129 
130  z = w = x + y;
131  SET_LOW_WORD(z,0);
132  v = y - (z - x);
133  t = a = -one / w;
134  SET_LOW_WORD(t,0);
135  s = one + t * z;
136  return t + a * (s + t * v);
137  }
138  }
139  }
140  }
141  if (ix >= 0x3FE59428) { /* |x| >= 0.6744 */
142  if (hx < 0) {
143  x = -x;
144  y = -y;
145  }
146  z = pio4 - x;
147  w = pio4lo - y;
148  x = z + w;
149  y = 0.0;
150  }
151  z = x * x;
152  w = z * z;
153  /*
154  * Break x^5*(T[1]+x^2*T[2]+...) into
155  * x^5(T[1]+x^4*T[3]+...+x^20*T[11]) +
156  * x^5(x^2*(T[2]+x^4*T[4]+...+x^22*[T12]))
157  */
158  r = T[1] + w * (T[3] + w * (T[5] + w * (T[7] + w * (T[9] +
159  w * T[11]))));
160  v = z * (T[2] + w * (T[4] + w * (T[6] + w * (T[8] + w * (T[10] +
161  w * T[12])))));
162  s = z * x;
163  r = y + z * (s * (r + v) + y);
164  r += T[0] * s;
165  w = x + r;
166  if (ix >= 0x3FE59428) {
167  v = (double) iy;
168  return (double) (1 - ((hx >> 30) & 2)) *
169  (v - 2.0 * (x - (w * w / (w + v) - r)));
170  }
171  if (iy == 1)
172  return w;
173  else {
174  /*
175  * if allow error up to 2 ulp, simply return
176  * -1.0 / (x+r) here
177  */
178  /* compute -1.0 / (x+r) accurately */
179  double a, t;
180  z = w;
181  SET_LOW_WORD(z,0);
182  v = r - (z - x); /* z+v = r+x */
183  t = a = -1.0 / w; /* a = -1.0/w */
184  SET_LOW_WORD(t,0);
185  s = 1.0 + t * z;
186  return t + a * (s + t * v);
187  }
188 }
