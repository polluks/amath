59 /**
60  * @brief Natural logarithm function (base e).
61  * @version 1.4
62  * @date 96/03/07
63  * @details
64  * <pre>
65  * Method :
66  * 1. Argument Reduction: find k and f such that
67  * x = 2^k * (1+f),
68  * where sqrt(2)/2 < 1+f < sqrt(2) .
69  *
70  * 2. Approximation of log(1+f).
71  * Let s = f/(2+f) ; based on log(1+f) = log(1+s) - log(1-s)
72  * = 2s + 2/3 s**3 + 2/5 s**5 + .....,
73  * = 2s + s*R
74  * We use a special Remes algorithm on [0,0.1716] to generate
75  * a polynomial of degree 14 to approximate R The maximum error
76  * of this polynomial approximation is bounded by 2**-58.45. In
77  * other words,
78  * 2 4 6 8 10 12 14
79  * R(z) ~ Lg1*s +Lg2*s +Lg3*s +Lg4*s +Lg5*s +Lg6*s +Lg7*s
80  * (the values of Lg1 to Lg7 are listed in the program)
81  * and
82  * | 2 14 | -58.45
83  * | Lg1*s +...+Lg7*s - R(z) | <= 2
84  * | |
85  * Note that 2s = f - s*f = f - hfsq + s*hfsq, where hfsq = f*f/2.
86  * In order to guarantee error in log below 1ulp, we compute log
87  * by
88  * log(1+f) = f - s*(f - R) (if f is not too large)
89  * log(1+f) = f - (hfsq - s*(hfsq+R)). (better accuracy)
90  *
91  * 3. Finally, log(x) = k*ln2 + log(1+f).
92  * = k*ln2_hi+(f-(hfsq-(s*(hfsq+R)+k*ln2_lo)))
93  * Here ln2 is split into two floating point number:
94  * ln2_hi + ln2_lo,
95  * where n*ln2_hi is always exact for |n| < 2000.
96  *
97  * Special cases:
98  * log(x) is NaN with signal if x < 0 (including -INF) ;
99  * log(+INF) is +INF; log(0) is -INF with signal;
100  * log(NaN) is that NaN with no signal.
101  *
102  * Accuracy:
103  * according to an error analysis, the error is always less than
104  * 1 ulp (unit in the last place).
105  *
106  * Constants:
107  * The hexadecimal values are the intended ones for the following
108  * constants. The decimal values may be used, provided that the
109  * compiler will convert from decimal to binary accurately enough
110  * to produce the hexadecimal values shown.
111  * </pre>
112  * @copyright Copyright (C) 1993 by Sun Microsystems, Inc. All rights reserved.
113  * @license Developed at SunSoft, a Sun Microsystems, Inc. business. Permission
114  * to use, copy, modify, and distribute this software is freely granted,
115  * provided that this notice is preserved.
116  */

118 double log(double x)
119 {
120  double hfsq,f,s,z,R,w,t1,t2,dk;
121  sword k,hx,i,j;
122  uword lx;
123 
124  EXTRACT_WORDS(hx,lx,x);
125 
126  k=0;
127  if (hx < 0x00100000) { /* x < 2**-1022 */
128  if (((hx&0x7fffffff)|lx)==0)
129  return -two54/zero; /* log(+-0)=-inf */
130  if (hx<0) return (x-x)/zero; /* log(-#) = NaN */
131  k -= 54;
132  x *= two54; /* subnormal number, scale up x */
133  GET_HIGH_WORD(hx,x); /* high word of x */
134  }
135  if (hx >= 0x7ff00000) return x+x;
136  k += (hx>>20)-1023;
137  hx &= 0x000fffff;
138  i = (hx+0x95f64)&0x100000;
139  SET_HIGH_WORD(x,hx|(i^0x3ff00000)); /* normalize x or x/2 */
140  k += (i>>20);
141  f = x-1.0;
142  if((0x000fffff&(2+hx))<3) { /* |f| < 2**-20 */
143  if(f==zero) {
144  if(k==0)
145  return zero;
146  else {
147  dk=(double)k;
148  return dk*ln2_hi+dk*ln2_lo;
149  }
150  }
151  R = f*f*(0.5-0.33333333333333333*f);
152  if(k==0) return f-R;
153  else {
154  dk=(double)k;
155  return dk*ln2_hi-((R-dk*ln2_lo)-f);
156  }
157  }
158  s = f/(2.0+f);
159  dk = (double)k;
160  z = s*s;
161  i = hx-0x6147a;
162  w = z*z;
163  j = 0x6b851-hx;
164  t1= w*(Lg2+w*(Lg4+w*Lg6));
165  t2= z*(Lg1+w*(Lg3+w*(Lg5+w*Lg7)));
166  i |= j;
167  R = t2+t1;
168  if(i>0) {
169  hfsq=0.5*f*f;
170  if(k==0) return f-(hfsq-s*(hfsq+R));
171  else
172  return dk*ln2_hi-((hfsq-(s*(hfsq+R)+dk*ln2_lo))-f);
173  } else {
174  if(k==0) return f-s*(f-R);
175  else
176  return dk*ln2_hi-((s*(f-R)-dk*ln2_lo)-f);
177  }
178 }
