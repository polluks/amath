45 /* double erf(double x)
46  * double erfc(double x)
47  * x
48  * 2 |\
49  * erf(x) = --------- | exp(-t*t)dt
50  * sqrt(pi) \|
51  * 0
52  *
53  * erfc(x) = 1-erf(x)
54  * Note that
55  * erf(-x) = -erf(x)
56  * erfc(-x) = 2 - erfc(x)
57  *
58  * Method:
59  * 1. For |x| in [0, 0.84375]
60  * erf(x) = x + x*R(x^2)
61  * erfc(x) = 1 - erf(x) if x in [-.84375,0.25]
62  * = 0.5 + ((0.5-x)-x*R) if x in [0.25,0.84375]
63  * where R = P/Q where P is an odd poly of degree 8 and
64  * Q is an odd poly of degree 10.
65  * -57.90
66  * | R - (erf(x)-x)/x | <= 2
67  *
68  *
69  * Remark. The formula is derived by noting
70  * erf(x) = (2/sqrt(pi))*(x - x^3/3 + x^5/10 - x^7/42 + ....)
71  * and that
72  * 2/sqrt(pi) = 1.128379167095512573896158903121545171688
73  * is close to one. The interval is chosen because the fix
74  * point of erf(x) is near 0.6174 (i.e., erf(x)=x when x is
75  * near 0.6174), and by some experiment, 0.84375 is chosen to
76  * guarantee the error is less than one ulp for erf.
77  *
78  * 2. For |x| in [0.84375,1.25], let s = |x| - 1, and
79  * c = 0.84506291151 rounded to single (24 bits)
80  * erf(x) = sign(x) * (c + P1(s)/Q1(s))
81  * erfc(x) = (1-c) - P1(s)/Q1(s) if x > 0
82  * 1+(c+P1(s)/Q1(s)) if x < 0
83  * |P1/Q1 - (erf(|x|)-c)| <= 2**-59.06
84  * Remark: here we use the taylor series expansion at x=1.
85  * erf(1+s) = erf(1) + s*Poly(s)
86  * = 0.845.. + P1(s)/Q1(s)
87  * That is, we use rational approximation to approximate
88  * erf(1+s) - (c = (single)0.84506291151)
89  * Note that |P1/Q1|< 0.078 for x in [0.84375,1.25]
90  * where
91  * P1(s) = degree 6 poly in s
92  * Q1(s) = degree 6 poly in s
93  *
94  * 3. For x in [1.25,1/0.35(~2.857143)],
95  * erfc(x) = (1/x)*exp(-x*x-0.5625+R1/S1)
96  * erf(x) = 1 - erfc(x)
97  * where
98  * R1(z) = degree 7 poly in z, (z=1/x^2)
99  * S1(z) = degree 8 poly in z
100  *
101  * 4. For x in [1/0.35,28]
102  * erfc(x) = (1/x)*exp(-x*x-0.5625+R2/S2) if x > 0
103  * = 2.0 - (1/x)*exp(-x*x-0.5625+R2/S2) if -6<x<0
104  * = 2.0 - tiny (if x <= -6)
105  * erf(x) = sign(x)*(1.0 - erfc(x)) if x < 6, else
106  * erf(x) = sign(x)*(1.0 - tiny)
107  * where
108  * R2(z) = degree 6 poly in z, (z=1/x^2)
109  * S2(z) = degree 7 poly in z
110  *
111  * Note1:
112  * To compute exp(-x*x-0.5625+R/S), let s be a single
113  * precision number and s := x; then
114  * -x*x = -s*s + (s-x)*(s+x)
115  * exp(-x*x-0.5626+R/S) =
116  * exp(-s*s-0.5625)*exp((s-x)*(s+x)+R/S);
117  * Note2:
118  * Here 4 and 5 make use of the asymptotic series
119  * exp(-x*x)
120  * erfc(x) ~ ---------- * ( 1 + Poly(1/x^2) )
121  * x*sqrt(pi)
122  * We use rational approximation to approximate
123  * g(s)=f(1/x^2) = log(erfc(x)*x) - x*x + 0.5625
124  * Here is the error bound for R1/S1 and R2/S2
125  * |R1/S1 - f(x)| < 2**(-62.57)
126  * |R2/S2 - f(x)| < 2**(-61.52)
127  *
128  * 5. For inf > x >= 28
129  * erf(x) = sign(x) *(1 - tiny) (raise inexact)
130  * erfc(x) = tiny*tiny (raise underflow) if x > 0
131  * = 2 - tiny if x<0
132  *
133  * 7. Special case:
134  * erf(0) = 0, erf(inf) = 1, erf(-inf) = -1,
135  * erfc(0) = 1, erfc(inf) = 0, erfc(-inf) = 2,
136  * erfc/erf(NaN) is NaN
137  */

224 double erf(x)
225 double x;
226 #endif
227 {
228  int hx,ix,i;
229  double R,S,P,Q,s,y,z,r;
230  hx = __HI(x);
231  ix = hx&0x7fffffff;
232  if(ix>=0x7ff00000) { /* erf(nan)=nan */
233  i = ((unsigned)hx>>31)<<1;
234  return (double)(1-i)+one/x; /* erf(+-inf)=+-1 */
235  }
236 
237  if(ix < 0x3feb0000) { /* |x|<0.84375 */
238  if(ix < 0x3e300000) { /* |x|<2**-28 */
239  if (ix < 0x00800000)
240  return 0.125*(8.0*x+efx8*x); /*avoid underflow */
241  return x + efx*x;
242  }
243  z = x*x;
244  r = pp0+z*(pp1+z*(pp2+z*(pp3+z*pp4)));
245  s = one+z*(qq1+z*(qq2+z*(qq3+z*(qq4+z*qq5))));
246  y = r/s;
247  return x + x*y;
248  }
249  if(ix < 0x3ff40000) { /* 0.84375 <= |x| < 1.25 */
250  s = fabs(x)-one;
251  P = pa0+s*(pa1+s*(pa2+s*(pa3+s*(pa4+s*(pa5+s*pa6)))));
252  Q = one+s*(qa1+s*(qa2+s*(qa3+s*(qa4+s*(qa5+s*qa6)))));
253  if(hx>=0) return erx + P/Q;
254  else return -erx - P/Q;
255  }
256  if (ix >= 0x40180000) { /* inf>|x|>=6 */
257  if(hx>=0) return one-tiny;
258  else return tiny-one;
259  }
260  x = fabs(x);
261  s = one/(x*x);
262  if(ix< 0x4006DB6E) { /* |x| < 1/0.35 */
263  R=ra0+s*(ra1+s*(ra2+s*(ra3+s*(ra4+s*(
264  ra5+s*(ra6+s*ra7))))));
265  S=one+s*(sa1+s*(sa2+s*(sa3+s*(sa4+s*(
266  sa5+s*(sa6+s*(sa7+s*sa8)))))));
267  } else { /* |x| >= 1/0.35 */
268  R=rb0+s*(rb1+s*(rb2+s*(rb3+s*(rb4+s*(
269  rb5+s*rb6)))));
270  S=one+s*(sb1+s*(sb2+s*(sb3+s*(sb4+s*(
271  sb5+s*(sb6+s*sb7))))));
272  }
273  z = x;
274  __LO(z) = 0;
275  r = __ieee754_exp(-z*z-0.5625)*__ieee754_exp((z-x)*(z+x)+R/S);
276  if(hx>=0) return one-r/x;
277  else return r/x-one;
278 }

281 double erfc(double x)
282 #else
283 double erfc(x)
284 double x;
285 #endif
286 {
287  int hx,ix;
288  double R,S,P,Q,s,y,z,r;
289  hx = __HI(x);
290  ix = hx&0x7fffffff;
291  if(ix>=0x7ff00000) { /* erfc(nan)=nan */
292  /* erfc(+-inf)=0,2 */
293  return (double)(((unsigned)hx>>31)<<1)+one/x;
294  }
295 
296  if(ix < 0x3feb0000) { /* |x|<0.84375 */
297  if(ix < 0x3c700000) /* |x|<2**-56 */
298  return one-x;
299  z = x*x;
300  r = pp0+z*(pp1+z*(pp2+z*(pp3+z*pp4)));
301  s = one+z*(qq1+z*(qq2+z*(qq3+z*(qq4+z*qq5))));
302  y = r/s;
303  if(hx < 0x3fd00000) { /* x<1/4 */
304  return one-(x+x*y);
305  } else {
306  r = x*y;
307  r += (x-half);
308  return half - r ;
309  }
310  }
311  if(ix < 0x3ff40000) { /* 0.84375 <= |x| < 1.25 */
312  s = fabs(x)-one;
313  P = pa0+s*(pa1+s*(pa2+s*(pa3+s*(pa4+s*(pa5+s*pa6)))));
314  Q = one+s*(qa1+s*(qa2+s*(qa3+s*(qa4+s*(qa5+s*qa6)))));
315  if(hx>=0) {
316  z = one-erx;
317  return z - P/Q;
318  } else {
319  z = erx+P/Q;
320  return one+z;
321  }
322  }
323  if (ix < 0x403c0000) { /* |x|<28 */
324  x = fabs(x);
325  s = one/(x*x);
326  if(ix< 0x4006DB6D) { /* |x| < 1/.35 ~ 2.857143*/
327  R=ra0+s*(ra1+s*(ra2+s*(ra3+s*(ra4+s*(
328  ra5+s*(ra6+s*ra7))))));
329  S=one+s*(sa1+s*(sa2+s*(sa3+s*(sa4+s*(
330  sa5+s*(sa6+s*(sa7+s*sa8)))))));
331  } else { /* |x| >= 1/.35 ~ 2.857143 */
332  if(hx<0&&ix>=0x40180000) return two-tiny;/* x < -6 */
333  R=rb0+s*(rb1+s*(rb2+s*(rb3+s*(rb4+s*(
334  rb5+s*rb6)))));
335  S=one+s*(sb1+s*(sb2+s*(sb3+s*(sb4+s*(
336  sb5+s*(sb6+s*sb7))))));
337  }
338  z = x;
339  __LO(z) = 0;
340  r = __ieee754_exp(-z*z-0.5625)*
341  __ieee754_exp((z-x)*(z+x)+R/S);
342  if(hx>0) return r/x;
343  else return two-r/x;
344  } else {
345  if(hx>0) return tiny*tiny;
346  else return two-tiny;
347  }
