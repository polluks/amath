31 /**
32  * @brief Inverse hyperbolic sine of complex number
33  * @version 1.0
34  * @date 14/09/15
35  * @details
36  * Inverse hyperbolic sine expressed using complex logarithms:
37  * <pre>
38  * asinh(z) = log(z + sqrt(z*z + 1))
39  *
40  * With branch cuts: -i INF to -i and i to i INF
41  *
42  * Domain: -INF to INF
43  * Range: -INF to INF
44  * </pre>
45  * More info is available at Wikipedia: <BR>
46  * http://en.wikipedia.org/wiki/Inverse_hyperbolic_function#Logarithmic_representation
47  */
48 complex casinh(complex z)
49 {
50  complex one = cpack(1.0, 0.0);
51  complex a = cadd(cmul(z, z), one);
52  complex b = cadd(z, csqrt(a));
53  complex w = clog(b);
54  return w;
55 }
